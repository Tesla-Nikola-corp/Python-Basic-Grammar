#예제
animal = '강아지'
name = '연탄이'
age = 4
hobby = '산책'
is_adult = age >= 3

print('우리집 ' + animal + '의 이름은 ' + name + '예요')
print(name + '는 ' + str(age) + '살이며, ' + hobby + '을 아주 좋아해요')
// 정수형은 print문에서 쓰이려면 str이라는 걸로 감싸줌
print(name + '는 어른일까요? ' + str(is_adult))
// Boolean형도 마찬가지

//----------------------------------------------------------

+가 아닌 쉼표로도 여러 문장을 합칠 수 있음.
이때는 정수형 변수와 Boolean 변수를 그대로 사용할 수 있음
쉼표가 들어가면 무조건 띄어쓰기가 한 칸 들어감
